package Ejercicio_2.EjercicioTodo;

import java.util.ArrayList;
import java.util.List;
import javax.swing.JOptionPane;

class Nodo {
    
    Integer valor;
    Nodo izquierdo;
    Nodo derecho;

    public Nodo(Integer valor) {
        this.valor = valor;
        this.izquierdo = null;
        this.derecho = null;
    }

    public Nodo() {
    }
    
    
    
    
    
}

class ArbolEnlazado {
    
    Nodo raiz;

    public ArbolEnlazado() {
        this.raiz = null;
    }
    
    public void insertar(int valor) {

        Nodo nuevoNodo = new Nodo(valor);
        if (raiz==null) {
            raiz=nuevoNodo;
        } else {
            Nodo auxiliar = raiz;
            Nodo padre;
            while(true) {
                padre=auxiliar; //se posiciona en la cabecera del nodo
                if (valor < auxiliar.valor) {
                    auxiliar=auxiliar.izquierdo; //si el valor es menor a la cabecera, se va a la izquierda, sino a la derecha
                    if (auxiliar==null) { //SI en la nueva posición no hay nada, se crea el nodo
                        padre.izquierdo=nuevoNodo;
                        return;
                    }                   
                } else {
                    auxiliar=auxiliar.derecho;
                    if (auxiliar==null) {
                        padre.derecho=nuevoNodo;
                        return;
                    }
                }
            }
        }

    }
    
    
    public List<Integer> recorrido(Nodo raiz,int opcion) {
        List<Integer> listado = new ArrayList<>();
        
        switch (opcion) {
            
            case 1:
                recorridoPreO(raiz, listado);
                break;
            case 2:
                recorridoPostO(raiz, listado);
                break;
            case 3:
                recorridoInO(raiz, listado);
                
            
        }
        
        return listado;
    }
    //Recorrido por preorden
    public void recorridoPreO(Nodo raiz, List<Integer> lista) {
            if (raiz != null) {
                lista.add(raiz.valor);
                recorridoPreO(raiz.izquierdo, lista); //por izquierda
                recorridoPreO(raiz.derecho, lista); //por derecha
            }
    }
    
    //Recorrido por postorden
    public void recorridoPostO(Nodo raiz, List<Integer> lista) {
        if (raiz != null) {
            recorridoPreO(raiz.izquierdo, lista); //por izquierda
            recorridoPreO(raiz.derecho, lista); //por derecha
            lista.add(raiz.valor);
        }
    }
    
    //Recorrido por preorden
    public void recorridoInO(Nodo raiz, List<Integer> lista) {
        if (raiz != null) {
            recorridoPreO(raiz.izquierdo, lista); //por izquierda
            lista.add(raiz.valor);
            recorridoPreO(raiz.derecho, lista); //por derecha
        }
    }
    
    public boolean busqueda(int valor) {  
        Nodo auxiliar = raiz;
        if (auxiliar.valor == valor) {
            return true;
        }
        try {
            while (auxiliar.valor != valor) {

                if (auxiliar.valor > valor) {
                    auxiliar = auxiliar.izquierdo;
                } else if (auxiliar.valor < valor) {
                    auxiliar = auxiliar.derecho;
                }
                if (auxiliar.valor == valor) {
                    return true;
                }

            }
        } catch (java.lang.NullPointerException e) {
            return false;
        }


        return false;

    }
    
    
    
}
